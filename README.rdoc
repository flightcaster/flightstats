= Usage

	require 'flightstats'
	
	FLIGHTSTATS_ACCOUNT_ID	= 'my_id'
	FLIGHTSTATS_USER_ID 	= "12345"
	FLIGHTSTATS_PASSWORD 	= "*****"
	
	# Displays the flight numbers of all the arriving flights

	end
